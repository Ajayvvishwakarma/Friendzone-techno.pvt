## ?? Quick Commands Reference

```bash
# START
net start MySQL80              # Start MySQL
mysql -u root -p               # Connect to MySQL
mysql -u root -p <setup_database.sql  # Create DB

# VERIFY
USE dashboard_db;              # Select database
SHOW TABLES;                    # Show all tables
SELECT * FROM users;           # View all users
SELECT COUNT(*) FROM users;    # Count users

# QUERY
SELECT * FROM users WHERE email = 'soni@gmail.com';  # Find user
INSERT INTO users (...) VALUES (...);  # Add user
UPDATE users SET firstName = 'X' WHERE ...;  # Update
DELETE FROM users WHERE ...;   # Delete

# DEBUG
DESCRIBE users;                # Table structure
SHOW CREATE TABLE users;       # Full schema
SELECT * FROM information_schema.tables;  # All tables

# EXIT
EXIT;                          # Exit MySQL
```
